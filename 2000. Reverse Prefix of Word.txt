2000. Reverse Prefix of Word

class Solution {
    public String reversePrefix(String word, char ch) {
       String s = "";
       int index = 0;
       for(int i=0;i<word.length();i++){
          if(word.charAt(i) == ch){
              index = i;
              break;
          }
       } 
       for(int i=index;i>=0;i--){
           s+=word.charAt(i);
       }
        for(int i=index+1;i<word.length();i++){
           s+=word.charAt(i);
       }
       return s;
    }
}
