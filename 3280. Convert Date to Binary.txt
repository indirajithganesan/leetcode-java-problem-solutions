3280. Convert Date to Binary

class Solution {
    public String convertDateToBinary(String date) {
        String arr[] = date.split("-");
        String b = "";
        for(int i=0;i<arr.length;i++){
            int a = Integer.parseInt(arr[i]);//2080
             b += Integer.toBinaryString(a);
             a = 0;
             if(i<arr.length-1){
             b+="-";
             }
        }
        return b;
    }
}
