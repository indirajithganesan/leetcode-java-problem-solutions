1637. Widest Vertical Area Between Two Points Containing No Points

import java.util.Arrays;

class Solution {
    public int maxWidthOfVerticalArea(int[][] points) {
        Arrays.sort(points, (a, b) -> Integer.compare(a[0], b[0]));//For sorting x coordinates
        int diff = 0;
        for (int i = 1; i < points.length; i++) {
            diff = Math.max(diff, points[i][0] - points[i - 1][0]);//For max difference
        }
        return diff;
    }
}
