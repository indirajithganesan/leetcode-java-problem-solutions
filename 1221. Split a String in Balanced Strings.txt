1221. Split a String in Balanced Strings

class Solution {
    public int balancedStringSplit(String s) {
        int count = 0;
        int isBalanced = 0;
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == 'R') {
                isBalanced++;
            } else {  // 'L'
                isBalanced--;
            }
            if (isBalanced == 0) {
                count++;
            }
        }
        return count;
    }
}
