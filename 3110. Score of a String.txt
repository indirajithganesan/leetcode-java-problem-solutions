3110. Score of a String

class Solution {
    public int scoreOfString(String s) {
        int a = 0;
        for(int i=0;i<s.length();i++){
            if(i+1<s.length()){
            int val = Math.abs(((int)s.charAt(i))-( (int)s.charAt(i+1) ));
             a += val;
        }
        }
        return a;
    }
}
