2824. Count Pairs Whose Sum is Less than Target

class Solution {
    public int countPairs(List<Integer> nums, int target) {
        int pair = 0;
     //   int sum = 0;
        for(int i=0;i<nums.size();i++){
            for(int j=0;j<nums.size();j++){
               if(nums.get(i)+nums.get(j)<target && i<j){
                  pair++;
               }
            }
        }
        return pair;
    }
}
